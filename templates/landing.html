# Authenticate
token = requests.post("/api/v1/auth/login/", json={
    "username": "you",

}).json()["access"]

# Create a report
report = requests.post(
    "/api/v1/reports/create/",
    headers={"Authorization": f"Bearer {token}"},
    json={"urls": "https://site.com\nhttps://site.com/about"}
).json()

print(report["id"])  # ready to poll
